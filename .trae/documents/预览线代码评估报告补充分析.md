**修复建议1: 实现布局与预览线的协调机制**

class LayoutPreviewLineCoordinator {
  constructor(layoutEngine, previewLineManager) {
    this.layoutEngine = layoutEngine;
    this.previewLineManager = previewLineManager;
    this.isLayoutInProgress = false;
  }
  
  async executeCoordinatedLayout() {
    this.isLayoutInProgress = true;
    
    // 1. 暂停预览线更新
    await this.previewLineManager.pauseAllUpdates();
    
    // 2. 执行布局
    const layoutResult = await this.layoutEngine.executeLayout();
    
    // 3. 基于新布局更新预览线
    await this.previewLineManager.updateBasedOnLayout(layoutResult);
    
    // 4. 恢复预览线更新
    await this.previewLineManager.resumeAllUpdates();
    
    this.isLayoutInProgress = false;
  }
}

class EventCoordinator extends EventEmitter {
  constructor() {
    super();
    this.eventQueue = [];
    this.isProcessing = false;
  }
  
  async coordinateEvent(eventType, eventData) {
    this.eventQueue.push({ type: eventType, data: eventData, timestamp: Date.now() });
    
    if (!this.isProcessing) {
      await this.processEventQueue();
    }
  }
  
  async processEventQueue() {
    this.isProcessing = true;
    
    while (this.eventQueue.length > 0) {
      const event = this.eventQueue.shift();
      await this.handleEvent(event);
    }
    
    this.isProcessing = false;
  }
}
