// ============================================
// 方案1: 主应用提供日志注入 (推荐)
// ============================================

// 在 portal-shell (主应用) 中创建日志服务
// src/services/SubAppLogger.ts

import { ref } from 'vue'

interface LogEntry {
  timestamp: string
  level: 'log' | 'warn' | 'error' | 'info'
  message: string
  appName: string
  source?: string
}

class SubAppLogger {
  private logs = ref<LogEntry[]>([])
  private appName = ''

  setAppName(name: string) {
    this.appName = name
  }

  // 收集日志
  log(level: LogEntry['level'], message: string, source?: string) {
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      appName: this.appName,
      source
    }
    this.logs.value.push(entry)
    
    // 同步到主应用
    if (window.parent !== window) {
      window.parent.postMessage({
        type: 'subapp-log',
        payload: entry
      }, '*')
    }
  }

  // 重写的 console 方法
  info(message: string, ...args: any[]) {
    this.log('info', message, args[0])
    console.info(`[${this.appName}]`, message, ...args)
  }

  warn(message: string, ...args: any[]) {
    this.log('warn', message, args[0])
    console.warn(`[${this.appName}]`, message, ...args)
  }

  error(message: string, ...args: any[]) {
    this.log('error', message, args[0])
    console.error(`[${this.appName}]`, message, ...args)
  }

  getLogs() {
    return this.logs
  }
}

export const subAppLogger = new SubAppLogger()
